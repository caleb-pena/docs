</ParamField>

<ParamField body="result" type="object or null">
</ParamField>

  <Expandable title="more">
    <ParamField body="score" type="float or null">

        </ParamField>
        <ParamField body="score" type="float or null">
          A value between 0 and 1 indicating how well the requirement was fulfilled.
        </ParamField>
        <ParamField body="reason" type="string or null">

        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

        <ParamField body="text" type="string or null">

        <ParamField body="score" type="float">
